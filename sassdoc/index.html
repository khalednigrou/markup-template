    mixins

    mq

    @mixin mq($size: md, $width: max) { ... }

    Parameters

    parameter Nameparameter Descriptionparameter Typeparameter Default value
    $size

    size in pixel

    		Number or Stringmd
    $width

    max|min

    		Numbermax

    Example

    Basic Usage

    // Example
.el {
    color: red;
    @include mq(){
        color: blue;
    }
}

// Output
.el { color: red; }
@media (max-width: 1020px) {
    .el { color: blue; }
}

    Advance Usage

    // Example
.el {
    color: red;
    @include mq(500, min){
        color: blue;
    }
}

// Output
.el { color: red; }
@media (max-width: 500px) {
    .el { color: blue; }
}
